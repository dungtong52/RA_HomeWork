create database db_game;
use db_game;

create table users
(
    id       int primary key auto_increment,
    username varchar(100) unique,
    password varchar(100),
    email    varchar(100),
    balance  double
);

create table seeds
(
    id         int primary key auto_increment,
    seeds_name varchar(100),
    price      double,
    image_url  varchar(255)
);

delimiter //

create procedure insert_user(
    in u_username varchar(100),
    in u_password varchar(100),
    in u_email varchar(100),
    in u_balance double
)
begin
    insert into users(username, password, email, balance)
    values (u_username, u_password, u_email, u_balance);
end //

create procedure check_username_exists(
    in u_username varchar(100)
)
begin
    select id from users where username = u_username;
end //

create procedure find_user(
    in u_username varchar(100),
    in u_password varchar(100)
)
begin
    select * from users where username = u_username and password = u_password;
end //

create procedure get_all_seeds()
begin
    select * from seeds;
end //

delimiter ;
